void PointLight(const float x, const float y, const float z, const float amb, const float diff, const float spec)
{
  glEnable(GL_LIGHTING);
  GLfloat light_ambient[] = { amb,amb,amb, 1.0 };
  GLfloat light_position[] = {-0.8,10.0,2.0, 0.0 };
  glLightfv(GL_LIGHT0, GL_AMBIENT, light_ambient);
  glLightfv(GL_LIGHT0, GL_POSITION, light_position);
  glEnable(GL_LIGHT0); //enable the light after setting the properties
}

void mouse(int button, int state, int x, int y){
	if(button == GLUT_LEFT_BUTTON && state == GLUT_DOWN){
	mousedown=true;
	xdiff=x-yrot;
	ydiff=-y+xrot;
	}
	else{
	mousedown = false;
	}
}
void mousemotion(int x, int y){
		if (mousedown){
		yrot=x-xdiff;
		xrot=y+ydiff;

		glutPostRedisplay();
		}
	}

void keyboard(unsigned char key, int x,int y)
{
	switch (key)
	{
	case 'w': // kedepan atau zoom
	case 'W':
		glTranslatef(0.0, 0.0, 3.0);
		break;
	case 'd': // geser kanan
	case 'D':
		glTranslatef(3.0, 0.0, 0.0);
		break;
	case 's': // kebelakang
	case 'S':
		glTranslatef(0.0, 0.0, -3.0);
		break;
	case 'a': // kekiri
	case 'A':
		glTranslatef(-3.0, 0.0, 0.0);
		break;
	case '7': // atas
		glTranslatef(0.0, 3.0, 0.0);
		break;
	case '9': //kebawah
		glTranslatef(0.0, -3.0, 0.0);
		break;
	case '2': //muter kebawah
		glRotatef(2.0, 1.0, 0.0, 0.0);
		break;
	case '8': // muter atas bawah bagian atas bawah
		glRotatef(-2.0, 1.0, 0.0, 0.0);
		break;
	case '6': // muter samping kanan
		glRotatef(2.0, 0.0, 1.0, 0.0);
		break;
	case '4': // muter samping kiri
		glRotatef(-2.0, 0.0, 1.0, 0.0);
		break;
	case '1': // muter atas bawah bagian atas bawah
		glRotatef(2.0, 0.0, 0.0, 1.0);
		break;
	case '3': // muter bawah atas bagian atas bawah
		glRotatef(-2.0, 0.0, 0.0, 1.0);
		break;
	case '5':
		if (is_depth)
		{
			is_depth = 0;
			glDisable(GL_DEPTH_TEST);
		}
		else
		{
			is_depth=1;
			glEnable(GL_DEPTH_TEST);
		}
	}
	tampil();
}

 void ukuran(int lebar, int tinggi)
 {
	if(tinggi == 0) tinggi = 1;

	glMatrixMode(GL_PROJECTION);
	glLoadIdentity();
	gluPerspective(50.0, lebar/tinggi, 5.0, 500.0);
	glTranslatef(0.0, -5.0, -150.0);
	glMatrixMode(GL_MODELVIEW);
 }

 int main(int argc, char**argv){
	glutInit(&argc,argv);
	glutInitDisplayMode(GLUT_SINGLE|GLUT_RGB);
	glutInitWindowSize(1020,420);
	glutInitWindowPosition(100,80);
	glutCreateWindow("3D");
	init();
	glutDisplayFunc(tampil);
	glutKeyboardFunc(keyboard);
	glutReshapeFunc(ukuran);
	glutMouseFunc(mouse);
	glutMotionFunc(mousemotion);
	glutMainLoop();
	return 0;
}
